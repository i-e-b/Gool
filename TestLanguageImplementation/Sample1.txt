fn main {
    k = 3;
    j = (10 + 1) * k;
    name = get_name(); # get user name in sub-function

    # Output a response, based on the input
    if name = "" {
        print("All right then, keep your secrets.\r\n");
    } else {
        print("Greetings, " + name + "!\r\n");
    }

    # Output numbers in a loop
    i = j / 3 - 11;
    loop countingLoop {
        i = i + 1;
        print(i);
        if i >= 5  { break countingLoop; }
    }
}

fn get_name {
    print("Hello. What is your name?\r\n");
    name = read_line();
    return name;
}
